# 执行追踪与可观测性

## 执行追踪

### 5.11 Agent 执行追踪与实时监控

#### 5.11.1 追踪层级

每个工作流执行自动生成一棵追踪树，三层结构：

```
Workflow Execution
  ├─ Agent Span (agent_name, stage, status, duration)
  │   ├─ LLM Call Span (model, prompt, response, tokens, duration)
  │   ├─ Tool Call Span (mcp_server, tool_name, args, result, duration)
  │   ├─ Knowledge Retrieval Span (query, kb_ids, result_count, duration)
  │   └─ Error Span (exception, stack_trace, retry_count)
  └─ Agent Span ...
```

#### 5.11.2 存储策略：文件为主 + DB 为索引

由于追踪树可能很大（多个 Agent、每个 Agent 多次 LLM/Tool 调用），全量存 SQLite 会导致表膨胀。采用**文件存追踪树 + DB 存元数据**的策略。

**文件**：追踪树完整 JSON 写入 workspace 目录，和产物放在一起：
```
/data/workflow_workspaces/{execution_id}/
├── trace.json           ← 完整追踪树
├── context.json
├── artifacts/
└── ...
```

**DB 表**：只存元数据，用于列表查询和快速过滤：

```sql
CREATE TABLE workflow_traces (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    execution_id VARCHAR(36) NOT NULL UNIQUE,
    workflow_id INTEGER REFERENCES workflows(id),
    app_id INTEGER REFERENCES apps(id),
    status VARCHAR(20),               -- running / success / failed / partial
    agent_count INTEGER DEFAULT 0,    -- Agent 数量
    total_duration_ms INTEGER,        -- 总耗时
    error_agent VARCHAR(100),         -- 失败 Agent 名称（如有）
    error_summary TEXT,               -- 错误摘要（1行，快速定位用）
    trace_file_path VARCHAR(500),     -- 追踪树文件路径
    started_at TIMESTAMP,
    completed_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_traces_execution ON workflow_traces(execution_id);
CREATE INDEX idx_traces_workflow ON workflow_traces(workflow_id);
CREATE INDEX idx_traces_status ON workflow_traces(status);
CREATE INDEX idx_traces_started ON workflow_traces(started_at);
```

#### 5.11.4 自动采集机制

通过 LangChain `BaseCallbackHandler` 自动采集，Agent 无需手动埋点：

```python
import time
import json
import os
from langchain.callbacks.base import BaseCallbackHandler
from loguru import logger

class ExecutionTracer:
    """执行追踪器 — 每个工作流执行一个实例"""

    def __init__(self, execution_id: str, workspace_path: str):
        self.execution_id = execution_id
        self.workspace_path = workspace_path
        self.trace = {
            "execution_id": execution_id,
            "workflow_name": "",
            "status": "running",
            "started_at": None,
            "spans": [],
            "errors": []
        }
        self._current_agent_span = None

    def start_execution(self, workflow_name: str):
        self.trace["started_at"] = self._now()
        self.trace["workflow_name"] = workflow_name
        self._flush()

    def start_agent(self, agent_name: str, agent_type: str, stage: int):
        span = {
            "span_id": f"span_agent_{stage}",
            "type": "agent",
            "agent_name": agent_name,
            "agent_type": agent_type,
            "stage": stage,
            "status": "running",
            "started_at": self._now(),
            "children": []
        }
        self.trace["spans"].append(span)
        self._current_agent_span = span
        self._flush()

    def add_llm_span(self, model: str, provider: str, prompt: str,
                      response: str, input_tokens: int, output_tokens: int,
                      duration_ms: int, status: str):
        span = {
            "span_id": self._next_span_id(),
            "type": "llm_call",
            "model": model, "provider": provider,
            "prompt": prompt, "response": response,
            "input_tokens": input_tokens, "output_tokens": output_tokens,
            "duration_ms": duration_ms, "status": status
        }
        self._current_agent_span["children"].append(span)
        self._flush()

    def add_tool_span(self, tool_name: str, mcp_server: str,
                       arguments: dict, result: str,
                       duration_ms: int, status: str):
        span = {
            "span_id": self._next_span_id(),
            "type": "tool_call",
            "tool_name": tool_name, "mcp_server": mcp_server,
            "arguments": arguments, "result": result,
            "duration_ms": duration_ms, "status": status
        }
        self._current_agent_span["children"].append(span)
        self._flush()

    def add_kb_span(self, query: str, kb_ids: list,
                     result_count: int, duration_ms: int):
        span = {
            "span_id": self._next_span_id(),
            "type": "knowledge_retrieval",
            "query": query, "kb_ids": kb_ids,
            "result_count": result_count,
            "duration_ms": duration_ms, "status": "success"
        }
        self._current_agent_span["children"].append(span)
        self._flush()

    def finish_agent(self, status: str, summary: str):
        agent = self._current_agent_span
        agent["status"] = status
        agent["duration_ms"] = self._elapsed_ms(agent["started_at"])
        agent["summary"] = summary
        agent["token_usage"] = self._sum_tokens(agent["children"])
        self._flush()

    def finish_execution(self, status: str):
        self.trace["status"] = status
        self.trace["completed_at"] = self._now()
        self.trace["total_duration_ms"] = self._elapsed_ms(
            self.trace["started_at"])
        self._flush()

    def _flush(self):
        """写入 trace.json 并推送 WebSocket 事件"""
        path = os.path.join(self.workspace_path, "trace.json")
        with open(path, "w", encoding="utf-8") as f:
            json.dump(self.trace, f, ensure_ascii=False, indent=2)

        # 推送实时更新
        from core.ws_manager import ws_manager
        asyncio.create_task(
            ws_manager.push(self.execution_id, {
                "type": "trace_update",
                "trace": self.trace
            })
        )


class TraceCallback(BaseCallbackHandler):
    """LangChain Callback → 自动生成追踪 span"""

    def __init__(self, tracer: ExecutionTracer, model: str, provider: str):
        self.tracer = tracer
        self.model = model
        self.provider = provider
        self._start_time = None
        self._prompt = None

    def on_llm_start(self, serialized, prompts, **kwargs):
        self._start_time = time.time()
        self._prompt = prompts[0] if prompts else ""

    def on_llm_end(self, response, **kwargs):
        duration_ms = int((time.time() - self._start_time) * 1000)
        usage = response.llm_output.get("token_usage", {})
        self.tracer.add_llm_span(
            model=self.model,
            provider=self.provider,
            prompt=self._prompt,
            response=response.generations[0][0].text,
            input_tokens=usage.get("prompt_tokens", 0),
            output_tokens=usage.get("completion_tokens", 0),
            duration_ms=duration_ms,
            status="success"
        )

    def on_tool_start(self, serialized, input_str, **kwargs):
        self._tool_start = time.time()

    def on_tool_end(self, output, **kwargs):
        duration_ms = int((time.time() - self._tool_start) * 1000)
        # 工具信息从 serialized 中提取
        self.tracer.add_tool_span(
            tool_name=kwargs.get("name", "unknown"),
            mcp_server=kwargs.get("mcp_server", ""),
            arguments=kwargs.get("arguments", {}),
            result=str(output),
            duration_ms=duration_ms,
            status="success"
        )
```

#### 5.11.5 WebSocket 实时推送

```python
from fastapi import WebSocket
from typing import Dict, Set
import json

class WorkflowWebSocketManager:
    """工作流执行 WebSocket 管理器"""

    def __init__(self):
        # execution_id → {websocket, ...}
        self._subscribers: Dict[str, Set[WebSocket]] = {}

    async def subscribe(self, execution_id: str, ws: WebSocket):
        await ws.accept()
        self._subscribers.setdefault(execution_id, set()).add(ws)

    def unsubscribe(self, execution_id: str, ws: WebSocket):
        subs = self._subscribers.get(execution_id)
        if subs:
            subs.discard(ws)

    async def push(self, execution_id: str, event: dict):
        """推送事件到所有订阅该 execution 的客户端"""
        subs = self._subscribers.get(execution_id, set())
        dead = set()
        for ws in subs:
            try:
                await ws.send_json(event)
            except Exception:
                dead.add(ws)
        subs -= dead

ws_manager = WorkflowWebSocketManager()


# FastAPI WebSocket 端点
@router.websocket("/ws/execution/{execution_id}/live")
async def watch_execution(websocket: WebSocket, execution_id: str):
    await ws_manager.subscribe(execution_id, websocket)
    try:
        while True:
            # 保持连接，等待客户端消息（心跳或取消）
            await websocket.receive_text()
    except Exception:
        pass
    finally:
        ws_manager.unsubscribe(execution_id, websocket)
```

WebSocket 事件类型：

| 事件 | 触发时机 | payload |
|------|---------|---------|
| `execution_started` | 工作流启动 | `{execution_id, workflow_name, agent_count}` |
| `agent_started` | Agent 开始执行 | `{agent_name, stage, started_at}` |
| `span_completed` | 任一字 span 完成 | `{span_id, type, duration_ms, status}` |
| `agent_completed` | Agent 完成 | `{agent_name, status, summary, duration_ms}` |
| `execution_completed` | 工作流完成 | `{execution_id, status, total_duration_ms}` |
| `execution_error` | 工作流异常 | `{execution_id, error_agent, error_message}` |

### 9.1 指标体系

| 类别 | 指标 | 采集方式 | 用途 |
|------|------|---------|------|
| **工作流** | 执行次数、成功率、P50/P95/P99 耗时 | Loguru + 结构化日志 | 性能基线 |
| **Agent** | 调用次数、成功/失败/超时次数 | Agent 节点包装器 | 定位瓶颈 Agent |
| **LLM** | Token 用量（input/output）、调用延迟 | LangChain callback | 成本核算 |
| **MCP** | 工具调用次数、MCP Server 健康状态 | MCPClient 统计 | MCP 可用性 |
| **API** | QPS、HTTP 状态码分布 | FastAPI middleware | 流量监控 |

### 9.2 结构化日志

```python
import json
import time
from loguru import logger

def log_workflow_execution(workflow_id: int, app_id: int, 
                           session_id: str, duration_ms: int,
                           status: str, agent_results: dict,
                           error: str = None):
    """结构化记录工作流执行"""
    log_entry = {
        "type": "workflow_execution",     # 日志类型标识
        "workflow_id": workflow_id,
        "app_id": app_id,
        "session_id": session_id,
        "duration_ms": duration_ms,
        "status": status,                 # success / partial / failed
        "agent_count": len(agent_results),
        "agent_results": {
            name: {"status": "ok" if r else "failed"}
            for name, r in agent_results.items()
        },
        "error": error,
        "timestamp": time.time()
    }
    logger.info(json.dumps(log_entry, ensure_ascii=False))

def log_token_usage(session_id: str, agent_name: str, 
                    input_tokens: int, output_tokens: int, 
                    model: str):
    """记录 Token 用量"""
    logger.info(json.dumps({
        "type": "token_usage",
        "session_id": session_id,
        "agent": agent_name,
        "model": model,
        "input_tokens": input_tokens,
        "output_tokens": output_tokens,
        "total_tokens": input_tokens + output_tokens,
        "timestamp": time.time()
    }, ensure_ascii=False))
```

### 9.3 LangChain Callback 集成

from langchain.callbacks.base import BaseCallbackHandler
from loguru import logger
import time
import json

class ObservabilityCallback(BaseCallbackHandler):
    """LLM 调用可观测性回调"""
    
    def __init__(self, session_id: str, agent_name: str):
        self.session_id = session_id
        self.agent_name = agent_name
        self._start_time = None
    
    def on_llm_start(self, serialized, prompts, **kwargs):

            "type": "llm_call",

            "model": response.llm_output.get("model_name", "unknown"),
