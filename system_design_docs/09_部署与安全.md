## 七、部署方案

### 7.1 SQLite部署

SQLite 作为嵌入式数据库，无需独立服务，数据存储在单个文件中，部署简单。

#### 数据库文件位置

```
/data/agent_platform.db
```

#### 初始化脚本 (init_db.py)

```python
import sqlite3
import os

def init_database(db_path: str = "/data/agent_platform.db"):
    """初始化SQLite数据库"""
    
    # 确保目录存在
    os.makedirs(os.path.dirname(db_path), exist_ok=True)
    
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()
    
    # 创建系统配置表
    cursor.execute('''
    CREATE TABLE IF NOT EXISTS sys_configs (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        config_key VARCHAR(100) NOT NULL UNIQUE,
        config_value TEXT NOT NULL,
        config_type VARCHAR(20) DEFAULT 'string',
        description VARCHAR(200),
        updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
    ''')
    
    # 创建用户表
    cursor.execute('''
    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        username VARCHAR(50) NOT NULL UNIQUE,
        password_hash VARCHAR(255) NOT NULL,
        api_key VARCHAR(64) NOT NULL UNIQUE,
        role VARCHAR(20) DEFAULT 'user',
        email VARCHAR(100),
        avatar VARCHAR(200),
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        last_login TIMESTAMP
    )
    ''')
    
    # 创建Agent表
    cursor.execute('''
    CREATE TABLE IF NOT EXISTS agents (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name VARCHAR(50) NOT NULL UNIQUE,
        display_name VARCHAR(100),
        description TEXT,
        role VARCHAR(20) NOT NULL,
        agent_type VARCHAR(20) DEFAULT 'local',
        llm_config TEXT,
        http_config TEXT,
        system_prompt TEXT,
        mcp_servers TEXT DEFAULT '[]',
        skills TEXT DEFAULT '[]',
        custom_tools TEXT DEFAULT '[]',
        status VARCHAR(20) DEFAULT 'active',
        created_by INTEGER REFERENCES users(id),
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        updated_at TIMESTAMP
    )
    ''')
    
    # 创建工作流表
    cursor.execute('''
    CREATE TABLE IF NOT EXISTS workflows (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name VARCHAR(100) NOT NULL,
        description TEXT,
        flow_type VARCHAR(20) NOT NULL,
        supervisor_agent_id INTEGER REFERENCES agents(id),
        worker_agent_ids TEXT DEFAULT '[]',
        edges TEXT DEFAULT '[]',
        parallel_groups TEXT DEFAULT '[]',
        human_interrupts TEXT DEFAULT '[]',
        status VARCHAR(20) DEFAULT 'draft',
        version INTEGER DEFAULT 1,
        created_by INTEGER REFERENCES users(id),
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        updated_at TIMESTAMP
    )
    ''')
    
    # 创建应用表
    cursor.execute('''
    CREATE TABLE IF NOT EXISTS apps (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name VARCHAR(100) NOT NULL,
        description TEXT,
        workflow_id INTEGER NOT NULL REFERENCES workflows(id),
        workflow_version INTEGER DEFAULT 1,
        api_key VARCHAR(64) NOT NULL UNIQUE,
        rate_limit INTEGER DEFAULT 60,
        enabled BOOLEAN DEFAULT 1,
        created_by INTEGER REFERENCES users(id),
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        updated_at TIMESTAMP
    )
    ''')
    
    # 创建会话表
    cursor.execute('''
    CREATE TABLE IF NOT EXISTS sessions (
        id VARCHAR(36) PRIMARY KEY,
        app_id INTEGER REFERENCES apps(id),
        user_id VARCHAR(100),
        messages TEXT DEFAULT '[]',
        thread_state TEXT,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        updated_at TIMESTAMP,
        expired_at TIMESTAMP
    )
    ''')
    
    # 创建审计日志表
    cursor.execute('''
    CREATE TABLE IF NOT EXISTS audit_logs (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        user_id INTEGER REFERENCES users(id),
        action VARCHAR(50) NOT NULL,
        target_type VARCHAR(50),
        target_id INTEGER,
        request_id VARCHAR(36),
        ip_address VARCHAR(45),
        details TEXT,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
    ''')
    
    # 创建索引
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_users_username ON users(username)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_users_api_key ON users(api_key)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_agents_name ON agents(name)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_agents_status ON agents(status)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_workflows_status ON workflows(status)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_apps_api_key ON apps(api_key)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_sessions_app_id ON sessions(app_id)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_sessions_expired_at ON sessions(expired_at)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_audit_logs_user_id ON audit_logs(user_id)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_audit_logs_created_at ON audit_logs(created_at)')
    
    # 插入默认配置
    default_configs = [
        ('llm.default.provider', 'openai', 'string', '默认LLM提供商'),
        ('llm.default.model', 'gpt-4o-mini', 'string', '默认模型'),
        ('llm.default.temperature', '0.3', 'number', '默认温度参数'),
        ('llm.default.api_key', '', 'secret', '默认API密钥'),
        ('system.max_tokens', '4096', 'number', '最大token限制'),
        ('system.session_ttl', '3600', 'number', '会话过期时间(秒)'),
        ('system.rate_limit.default', '60', 'number', '默认限流(次/分钟)')
    ]
    
    for config in default_configs:
        cursor.execute('''
        INSERT OR IGNORE INTO sys_configs (config_key, config_value, config_type, description)
        VALUES (?, ?, ?, ?)
        ''', config)
    
    conn.commit()
    conn.close()
    print(f"数据库初始化完成: {db_path}")

if __name__ == "__main__":
    init_database()
```

### 7.3 备份策略

SQLite 数据库备份简单，直接复制数据库文件即可：

```bash
# 备份脚本 backup.sh
#!/bin/bash
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backup"
DB_FILE="/data/agent_platform.db"

# 创建备份目录
mkdir -p $BACKUP_DIR

# 使用 SQLite 在线备份 API（推荐）
sqlite3 $DB_FILE ".backup '${BACKUP_DIR}/agent_platform_${DATE}.db'"

# 或者简单复制
# cp $DB_FILE ${BACKUP_DIR}/agent_platform_${DATE}.db

# 保留最近7天的备份
find $BACKUP_DIR -name "agent_platform_*.db" -mtime +7 -delete

echo "Backup completed: agent_platform_${DATE}.db"
```

定时任务（crontab）：
```bash
# 每天凌晨2点备份
0 2 * * * /scripts/backup.sh >> /logs/backup.log 2>&1
```


## 八、安全设计

### 8.1 API Key 加密存储

敏感字段（LLM API Key、应用 API Key 等）使用 Fernet 对称加密方案：

```python
from cryptography.fernet import Fernet
import os

class CryptoManager:
    """加密管理器"""
    
    def __init__(self):
        # 加密密钥从环境变量读取，不存在时自动生成
        self._key = os.getenv("FERNET_KEY")
        if not self._key:
            self._key = Fernet.generate_key().decode()
            os.environ["FERNET_KEY"] = self._key
        self._fernet = Fernet(self._key.encode() if isinstance(self._key, str) else self._key)
    
    def encrypt(self, plaintext: str) -> str:
        """加密"""
        return self._fernet.encrypt(plaintext.encode()).decode()
    
    def decrypt(self, ciphertext: str) -> str:
        """解密"""
        return self._fernet.decrypt(ciphertext.encode()).decode()

# 使用示例
crypto = CryptoManager()
encrypted_key = crypto.encrypt("sk-xxxx")  # 存储到数据库
decrypted_key = crypto.decrypt(encrypted_key)  # 运行时使用
```

**密钥管理规则**：
| 项目 | 策略 |
|------|------|
| 加密算法 | Fernet (AES-128-CBC + HMAC) |
| 密钥存储 | 环境变量 `FERNET_KEY`，不写入代码或数据库 |
| 密钥轮换 | 提供 `POST /system/crypto/rotate` API，用新密钥重新加密所有敏感字段 |
| 日志脱敏 | `loguru` 配置 `patch` 函数，自动替换 `sk-*`、`Bearer *` 等模式为 `***` |

### 8.2 数据库文件安全

```
正确路径: /data/agent_platform.db      (独立数据目录)
错误路径: ./backend/data/agent_platform.db  (在 web 可访问路径下)
```

- 数据库文件禁止放在源码目录或 web 静态资源目录下
- Docker 部署时通过 volume 挂载到宿主机，确保容器重建后数据不丢失

### 8.3 Prompt 注入风险

| 风险场景 | 缓解措施 |
|------|------|
| 用户在 Web UI 编辑 System Prompt 时注入恶意指令 | System Prompt 编辑需管理员权限，变更记录审计日志 |
| 外部用户通过 Chat API 注入指令覆盖 System Prompt | Chat API 入口对用户输入做角色隔离（user message 与 system message 分开注入） |
| MCP 工具返回值包含指令注入 | MCP 工具输出标记为 `tool_output` 角色，LLM 应区分对待 |
| Skill 文件被篡改 | Skill 文件路径只读，仅管理员可更新 |

### 8.4 API 安全

| 项目 | 措施 |
|------|------|
| 管理 API 认证 | JWT Token（7天过期）+ 角色鉴权 |
| Chat API 认证 | `X-API-Key` Header 或 `Authorization: Bearer` |
| 限流 | Redis 令牌桶，默认 60次/分钟，可按应用配置 |
| CORS | 生产环境白名单，开发环境允许 `localhost` |
| 请求大小限制 | 最大 10MB（FastAPI middleware） |


## 九、可观测性设计

### 9.1 指标体系

| 类别 | 指标 | 采集方式 | 用途 |
|------|------|---------|------|
| **工作流** | 执行次数、成功率、P50/P95/P99 耗时 | Loguru + 结构化日志 | 性能基线 |
| **Agent** | 调用次数、成功/失败/超时次数 | Agent 节点包装器 | 定位瓶颈 Agent |
| **LLM** | Token 用量（input/output）、调用延迟 | LangChain callback | 成本核算 |
| **MCP** | 工具调用次数、MCP Server 健康状态 | MCPClient 统计 | MCP 可用性 |
| **API** | QPS、HTTP 状态码分布 | FastAPI middleware | 流量监控 |

### 9.2 结构化日志

```python
import json
import time
from loguru import logger

def log_workflow_execution(workflow_id: int, app_id: int, 
                           session_id: str, duration_ms: int,
                           status: str, agent_results: dict,
                           error: str = None):
    """结构化记录工作流执行"""
    log_entry = {
        "type": "workflow_execution",     # 日志类型标识
        "workflow_id": workflow_id,
        "app_id": app_id,
        "session_id": session_id,
        "duration_ms": duration_ms,
        "status": status,                 # success / partial / failed
        "agent_count": len(agent_results),
        "agent_results": {
            name: {"status": "ok" if r else "failed"}
            for name, r in agent_results.items()
        },
        "error": error,
        "timestamp": time.time()
    }
    logger.info(json.dumps(log_entry, ensure_ascii=False))

def log_token_usage(session_id: str, agent_name: str, 
                    input_tokens: int, output_tokens: int, 
                    model: str):
    """记录 Token 用量"""
    logger.info(json.dumps({
        "type": "token_usage",
        "session_id": session_id,
        "agent": agent_name,
        "model": model,
        "input_tokens": input_tokens,
        "output_tokens": output_tokens,
        "total_tokens": input_tokens + output_tokens,
        "timestamp": time.time()
    }, ensure_ascii=False))
```

### 9.3 LangChain Callback 集成

```python
from langchain.callbacks.base import BaseCallbackHandler
from loguru import logger
import time
import json

class ObservabilityCallback(BaseCallbackHandler):
    """LLM 调用可观测性回调"""
    
    def __init__(self, session_id: str, agent_name: str):
        self.session_id = session_id
        self.agent_name = agent_name
        self._start_time = None
    
    def on_llm_start(self, serialized, prompts, **kwargs):
        self._start_time = time.time()
    
    def on_llm_end(self, response, **kwargs):
        duration_ms = int((time.time() - self._start_time) * 1000)
        token_usage = response.llm_output.get("token_usage", {})
        
        logger.info(json.dumps({
            "type": "llm_call",
            "session_id": self.session_id,
            "agent": self.agent_name,
            "model": response.llm_output.get("model_name", "unknown"),
            "duration_ms": duration_ms,
            "input_tokens": token_usage.get("prompt_tokens", 0),
            "output_tokens": token_usage.get("completion_tokens", 0)
        }, ensure_ascii=False))
```
