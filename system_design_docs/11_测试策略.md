### 11.4 MCP Server Mock

```python
import asyncio
import json
from unittest.mock import AsyncMock

class MockMCPServer:
    """模拟 MCP Server 用于测试"""
    
    def __init__(self, tools: list[dict]):
        self.tools = tools
    
    async def serve(self, host="localhost", port=0):
        """启动模拟 MCP Server"""
        server = await asyncio.start_server(
            self._handle_client, host, port
        )
        return server
    
    async def _handle_client(self, reader, writer):
        """处理 JSON-RPC 请求"""
        while True:
            line = await reader.readline()
            if not line:
                break
            request = json.loads(line.decode())
            
            method = request.get("method", "")
            
            if method == "initialize":
                response = self._init_response(request)
            elif method == "tools/list":
                response = self._list_tools_response(request)
            elif method == "tools/call":
                response = self._call_tool_response(request)
            else:
                response = self._error_response(request, -32601, "Method not found")
            
            writer.write((json.dumps(response) + "\n").encode())
            await writer.drain()
    
    def _init_response(self, request):
        return {
            "jsonrpc": "2.0",
            "id": request["id"],
            "result": {
                "protocolVersion": "2024-11-05",
                "capabilities": {"tools": {}}
            }
        }
    
    def _list_tools_response(self, request):
        return {
            "jsonrpc": "2.0",
            "id": request["id"],
            "result": {"tools": self.tools}
        }
    
    def _call_tool_response(self, request):
        tool_name = request["params"]["name"]
        # 返回预定义的模拟响应
        return {
            "jsonrpc": "2.0",
            "id": request["id"],
            "result": {
                "content": [{"type": "text", "text": f"mock_result: {tool_name}"}]
            }
        }
```
