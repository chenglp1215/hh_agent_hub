# 认证模块详细设计

## 一、认证体系概览

本系统有两套独立的认证通道：

| 通道 | 适用对象 | 认证方式 | 过期策略 |
|------|---------|---------|---------|
| **管理端** | Web 控制台用户 | JWT (Bearer Token) | 7 天过期, 支持刷新 |
| **Chat API** | 外部调用者 | API Key (Header) | 永久有效, 支持轮换 |

## 二、JWT 认证 (管理端)

### 2.1 登录流程

```
POST /api/v1/auth/login
  Body: { username, password }
    │
    ▼
验证密码 (bcrypt)
    │
    ├── 失败 → 401 { code: -1, message: "用户名或密码错误" }
    │
    └── 成功 → 生成 access_token + refresh_token
                   │
                   ▼
              更新 users.last_login
                   │
                   ▼
              返回 { access_token, refresh_token, user: { id, username, role } }
```

### 2.2 Token 格式

```json
// access_token payload
{
  "sub": "1",                    // user.id
  "username": "admin",
  "role": "admin",
  "type": "access",
  "exp": 1750320000,             // 7 天后
  "iat": 1749715200
}

// refresh_token payload
{
  "sub": "1",
  "type": "refresh",
  "exp": 1752912000,             // 30 天后
  "iat": 1749715200
}
```

### 2.3 实现代码

```python
# utils/jwt.py
import jwt
from datetime import datetime, timedelta
from config import settings

def create_access_token(user_id: int, username: str, role: str) -> str:
    payload = {
        "sub": str(user_id),
        "username": username,
        "role": role,
        "type": "access",
        "exp": datetime.utcnow() + timedelta(days=7),
        "iat": datetime.utcnow()
    }
    return jwt.encode(payload, settings.SECRET_KEY, algorithm="HS256")

def create_refresh_token(user_id: int) -> str:
    payload = {
        "sub": str(user_id),
        "type": "refresh",
        "exp": datetime.utcnow() + timedelta(days=30),
        "iat": datetime.utcnow()
    }
    return jwt.encode(payload, settings.SECRET_KEY, algorithm="HS256")

def decode_token(token: str) -> dict:
    return jwt.decode(token, settings.SECRET_KEY, algorithms=["HS256"])
```

### 2.4 依赖注入

```python
# api/deps.py
from fastapi import Depends, HTTPException, Header
from utils.jwt import decode_token
from models.user import User

async def get_current_user(authorization: str = Header(...)) -> User:
    """从 JWT 提取当前用户"""
    if not authorization.startswith("Bearer "):
        raise HTTPException(401, "Invalid authorization header")
    token = authorization[7:]
    try:
        payload = decode_token(token)
        if payload.get("type") != "access":
            raise HTTPException(401, "Invalid token type")
    except jwt.ExpiredSignatureError:
        raise HTTPException(401, "Token expired")
    except jwt.InvalidTokenError:
        raise HTTPException(401, "Invalid token")

    user = await User.get_or_none(id=int(payload["sub"]))
    if not user:
        raise HTTPException(401, "User not found")
    return user

async def require_admin(user: User = Depends(get_current_user)) -> User:
    """仅 admin 可访问"""
    if user.role != "admin":
        raise HTTPException(403, "Admin required")
    return user
```

### 2.5 API 端点

```python
# api/auth.py
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from utils.crypto import crypto
from utils.jwt import create_access_token, create_refresh_token, decode_token
from models.user import User

router = APIRouter(prefix="/api/v1/auth")

class LoginRequest(BaseModel):
    username: str
    password: str

class TokenResponse(BaseModel):
    access_token: str
    refresh_token: str
    user: dict

@router.post("/login", response_model=TokenResponse)
async def login(req: LoginRequest):
    user = await User.get_or_none(username=req.username)
    if not user or not crypto.verify_password(req.password, user.password_hash):
        raise HTTPException(401, "用户名或密码错误")

    user.last_login = datetime.utcnow()
    await user.save()

    return {
        "access_token": create_access_token(user.id, user.username, user.role),
        "refresh_token": create_refresh_token(user.id),
        "user": {"id": user.id, "username": user.username, "role": user.role}
    }

@router.post("/refresh")
async def refresh_token(refresh_token: str = Header(...)):
    try:
        payload = decode_token(refresh_token)
        if payload.get("type") != "refresh":
            raise HTTPException(401)
    except Exception:
        raise HTTPException(401, "Invalid refresh token")

    user = await User.get_or_none(id=int(payload["sub"]))
    if not user:
        raise HTTPException(401, "User not found")

    return {
        "access_token": create_access_token(user.id, user.username, user.role),
        "refresh_token": create_refresh_token(user.id)
    }

@router.post("/register")
async def register(req: LoginRequest):
    """仅开发期开放，生产环境需 admin 权限"""
    existing = await User.get_or_none(username=req.username)
    if existing:
        raise HTTPException(400, "用户名已存在")

    import secrets
    user = await User.create(
        username=req.username,
        password_hash=crypto.hash_password(req.password),
        api_key=f"sk-{secrets.token_urlsafe(32)}",
        role="user"
    )
    return {"id": user.id, "username": user.username}
```

## 三、API Key 认证 (Chat API)

### 3.1 验证中间件

```python
# api/middleware.py
from fastapi import Request, HTTPException
from starlette.middleware.base import BaseHTTPMiddleware
from models.app import App

class ApiKeyMiddleware(BaseHTTPMiddleware):
    """Chat API 专用 — 验证 X-API-Key"""

    CHAT_PATHS = ["/api/v1/chat"]  # 仅 Chat 路由走 API Key

    async def dispatch(self, request: Request, call_next):
        if not any(request.url.path.startswith(p) for p in self.CHAT_PATHS):
            return await call_next(request)  # 非 Chat 路由跳过

        api_key = request.headers.get("X-API-Key") or \
                  (request.headers.get("Authorization", "").replace("Bearer ", ""))

        if not api_key:
            return JSONResponse({"code": -1, "message": "Missing API Key"}, 401)

        app = await App.get_or_none(api_key=api_key, enabled=True)
        if not app:
            return JSONResponse({"code": -1, "message": "Invalid API Key"}, 401)

        request.state.app = app  # 注入到请求上下文
        return await call_next(request)
```

### 3.2 API Key 轮换

```python
# api/apps.py
@router.post("/apps/{id}/rotate-key")
async def rotate_api_key(id: int, user: User = Depends(require_admin)):
    app = await App.get_or_none(id=id)
    if not app:
        raise HTTPException(404)

    import secrets
    new_key = f"sk-{secrets.token_urlsafe(32)}"
    app.api_key = new_key
    await app.save()

    return {"api_key": new_key}  # 旧 key 立即失效
```

## 四、密码策略

```python
# utils/crypto.py
import bcrypt
from cryptography.fernet import Fernet

class CryptoManager:
    def __init__(self):
        key = os.getenv("FERNET_KEY") or Fernet.generate_key().decode()
        self._fernet = Fernet(key.encode() if isinstance(key, str) else key)

    def hash_password(self, password: str) -> str:
        return bcrypt.hashpw(password.encode(), bcrypt.gensalt()).decode()

    def verify_password(self, password: str, hashed: str) -> bool:
        return bcrypt.checkpw(password.encode(), hashed.encode())

    def encrypt(self, plaintext: str) -> str:
        return self._fernet.encrypt(plaintext.encode()).decode()

    def decrypt(self, ciphertext: str) -> str:
        return self._fernet.decrypt(ciphertext.encode()).decode()

crypto = CryptoManager()
```

## 五、权限矩阵

| 操作 | 未登录 | user | admin |
|------|--------|------|-------|
| 查看 Agent 列表 | ❌ | ✅ | ✅ |
| 查看 MCP Server 列表 | ❌ | ✅ | ✅ |
| 创建/编辑/删除 Agent | ❌ | ✅ | ✅ |
| 注册/编辑/删除 MCP Server | ❌ | ❌ | ✅ |
| 创建/编辑/删除知识库 | ❌ | ❌ | ✅ |
| 查看系统配置 | ❌ | ❌ | ✅ |
| 修改系统配置 | ❌ | ❌ | ✅ |
| 调用 Chat API | API Key | API Key | API Key |
| WebSocket 追踪 | ❌ | ✅ | ✅ |

**设计说明**：
- user 可以创建和管理自己的 Agent（日常使用）
- admin 管理平台级资源（MCP Server、知识库、系统配置）
- Chat API 不经过 Web 登录，用独立的 API Key 体系

## 六、前端认证流程

```
App 启动
  │
  ├── 已存储 token → 尝试请求 /auth/me 验证有效性
  │     ├── 有效 → 进入主控台
  │     └── 过期 → 尝试 refresh
  │           ├── 成功 → 更新 token → 进入主控台
  │           └── 失败 → 跳转登录页
  │
  └── 无 token → 跳转登录页
```

```typescript
// stores/user.ts
export const useUserStore = defineStore('user', () => {
  const user = ref(null)
  const token = ref(localStorage.getItem('access_token'))

  async function login(username: string, password: string) {
    const res = await api.post('/auth/login', { username, password })
    token.value = res.access_token
    localStorage.setItem('access_token', res.access_token)
    localStorage.setItem('refresh_token', res.refresh_token)
    user.value = res.user
  }

  async function refreshToken() {
    const refresh = localStorage.getItem('refresh_token')
    if (!refresh) throw new Error('No refresh token')
    const res = await api.post('/auth/refresh', null, {
      headers: { 'refresh-token': refresh }
    })
    token.value = res.access_token
    localStorage.setItem('access_token', res.access_token)
  }

  function logout() {
    token.value = null
    user.value = null
    localStorage.clear()
    router.push('/login')
  }

  // Axios 拦截器: 自动附加 Authorization header
  // 401 时自动 refresh，refresh 失败跳登录

  return { user, token, login, refreshToken, logout }
})
```
