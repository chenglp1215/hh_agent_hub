# 研发团队多Agent协同平台 - 系统设计文档（SQLite版）

## 一、概述

### 1.1 项目定位
面向研发团队内部使用的多Agent协同平台，支持配置多种LLM、创建专属Agent、编排多Agent协作流程，并通过统一API对外提供服务。

### 1.2 核心功能
| 模块 | 功能 |
|------|------|
| 系统配置 | LLM配置、用户管理 |
| Agent管理 | 创建/编辑Agent、配置LLM/MCP/Skills/Prompt |
| 工作流编排 | 多Agent协同流程设计（监督者/流水线/图模式） |
| 应用管理 | 关联工作流、生成对外API、会话管理 |
| 对话测试 | 在线调试Agent/工作流 |

### 1.3 技术栈
| 层级 | 技术选型 |
|------|---------|
| 后端框架 | Python 3.11 + FastAPI |
| Agent编排 | LangGraph |
| LLM接入 | LangChain |
| 数据库 | SQLite（开发/小规模生产） |
| 缓存/会话 | Redis（可选，生产环境推荐） |

### 5.4 LLM管理器 llm_manager.py

```python
from typing import Dict, Any, Optional
from langchain_openai import ChatOpenAI
from langchain_anthropic import ChatAnthropic
from langchain_core.language_models import BaseChatModel

class LLMManager:
    """LLM实例管理器"""
    
    def __init__(self):
        self._instances: Dict[str, BaseChatModel] = {}
    
    def get_llm(self, llm_config: Dict[str, Any]) -> BaseChatModel:
        """根据配置获取LLM实例"""
        cache_key = f"{llm_config['provider']}_{llm_config['model']}"
        
        if cache_key not in self._instances:
            self._instances[cache_key] = self._create_llm(llm_config)
        
        return self._instances[cache_key]
    
    def _create_llm(self, config: Dict[str, Any]) -> BaseChatModel:
        provider = config.get("provider", "openai")
        model = config.get("model", "gpt-4o-mini")
        temperature = config.get("temperature", 0.3)
        max_tokens = config.get("max_tokens", 4096)
        api_key = config.get("api_key")
        base_url = config.get("base_url")
        
        if provider == "openai":
            return ChatOpenAI(
                model=model,
                temperature=temperature,
                max_tokens=max_tokens,
                api_key=api_key,
                base_url=base_url
            )
        elif provider == "anthropic":
            return ChatAnthropic(
                model=model,
                temperature=temperature,
                max_tokens=max_tokens,
                api_key=api_key,
                base_url=base_url
            )
        elif provider == "ollama":
            from langchain_community.chat_models import ChatOllama
            return ChatOllama(
                model=model,
                temperature=temperature,
                base_url=base_url or "http://localhost:11434"
            )
        else:
            raise ValueError(f"Unsupported provider: {provider}")
    
    def clear_cache(self):
        """清除缓存"""
        self._instances.clear()

llm_manager = LLMManager()
```

### 5.5 Agent节点工厂 agent_factory.py

```python
import json
import httpx
from typing import Dict, Any, Callable, Optional
from langgraph.prebuilt import create_react_agent
from langchain_core.tools import tool
from core.llm_manager import llm_manager

class HttpAgentClient:
    """HTTP Agent客户端 - 对接外部Agent Chat HTTP API"""
    
    def __init__(self, http_config: Dict[str, Any]):
        self.base_url = http_config.get("base_url", "").rstrip("/")
        self.endpoint = http_config.get("endpoint", "/chat")
        self.method = http_config.get("method", "POST").upper()
        self.headers = http_config.get("headers", {})
        self.timeout = http_config.get("timeout", 30)
        self.retry_config = http_config.get("retry", {})
        self.request_template = http_config.get("request_template")
        self.response_mapping = http_config.get("response_mapping", {})
    
    def _build_request_body(self, user_input: str, session_id: str, context: Dict[str, Any]) -> Dict[str, Any]:
        """构建请求体"""
        if self.request_template:
            # 使用模板渲染请求体
            body = {}
            for key, value in self.request_template.items():
                if isinstance(value, str):
                    body[key] = value.replace("{{user_input}}", user_input) \
                                     .replace("{{session_id}}", session_id) \
                                     .replace("{{context}}", json.dumps(context))
                else:
                    body[key] = value
            return body
        else:
            # 默认请求格式
            return {
                "message": user_input,
                "session_id": session_id,
                "context": context
            }
    
    def _parse_response(self, response_data: Dict[str, Any]) -> str:
        """解析响应数据"""
        if self.response_mapping:
            output_field = self.response_mapping.get("output_field", "response.content")
            # 支持嵌套字段访问，如 "response.content"
            result = response_data
            for field in output_field.split("."):
                if isinstance(result, dict):
                    result = result.get(field, "")
                else:
                    result = ""
                    break
            return str(result)
        else:
            # 默认从 response.content 或 content 或直接返回
            if "response" in response_data:
                return response_data["response"].get("content", str(response_data["response"]))
            return response_data.get("content", str(response_data))
    
    async def invoke(self, user_input: str, session_id: str, context: Dict[str, Any] = None) -> str:
        """调用外部Agent HTTP API"""
        url = f"{self.base_url}{self.endpoint}"
        body = self._build_request_body(user_input, session_id, context or {})
        
        max_attempts = self.retry_config.get("max_attempts", 3) if self.retry_config else 1
        delay_ms = self.retry_config.get("delay_ms", 1000) if self.retry_config else 1000
        
        last_error = None
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            for attempt in range(max_attempts):
                try:
                    if self.method == "POST":
                        response = await client.post(url, json=body, headers=self.headers)
                    elif self.method == "GET":
                        response = await client.get(url, params=body, headers=self.headers)
                    else:
                        raise ValueError(f"Unsupported HTTP method: {self.method}")
                    
                    response.raise_for_status()
                    data = response.json()
                    return self._parse_response(data)
                    
                except httpx.HTTPError as e:
                    last_error = e
                    if attempt < max_attempts - 1:
                        import asyncio
                        await asyncio.sleep(delay_ms / 1000)
                    continue
        
        raise Exception(f"HTTP Agent调用失败: {last_error}")


class AgentNodeFactory:
    """Agent节点工厂 - 支持local和http两种Agent类型"""
    
    def __init__(self):
        self._nodes = {}
        self._http_clients: Dict[int, HttpAgentClient] = {}
    
    def create_node(self, agent_config: Dict[str, Any]) -> Callable:
        """创建Agent节点函数"""
        agent_id = agent_config.get("id")
        if agent_id in self._nodes:
            return self._nodes[agent_id]
        
        agent_type = agent_config.get("agent_type", "local")
        
        if agent_type == "http":
            return self._create_http_agent_node(agent_config)
        else:
            return self._create_local_agent_node(agent_config)
    
    def _create_local_agent_node(self, agent_config: Dict[str, Any]) -> Callable:
        """创建本地Agent节点（使用LangChain + LangGraph）"""
        agent_id = agent_config.get("id")
        
        # 1. 获取LLM
        llm_config = agent_config.get("llm_config", {})
        if not llm_config:
            raise ValueError(f"Local Agent {agent_id} 缺少 llm_config")
        llm = llm_manager.get_llm(llm_config)
        
        # 2. 加载工具（MCP + 自定义）
        tools = self._load_tools(agent_config)
        
        # 3. 构建完整提示词
        full_prompt = self._build_prompt(agent_config)
        
        # 4. 创建ReAct Agent
        react_agent = create_react_agent(
            model=llm,
            tools=tools,
            prompt=full_prompt
        )
        
        # 5. 定义节点函数
        def agent_node(state: Dict[str, Any]) -> Dict[str, Any]:
            result = react_agent.invoke({
                "messages": state.get("messages", []),
                "user_input": state.get("user_input", "")
            })
            return {
                "messages": result.get("messages", []),
                "intermediate_results": {
                    **state.get("intermediate_results", {}),
                    agent_config.get("name", "unknown"): result.get("output", "")
                }
            }
        
        self._nodes[agent_id] = agent_node
        return agent_node
    
    def _create_http_agent_node(self, agent_config: Dict[str, Any]) -> Callable:
```

### 5.6 工作流引擎 workflow_engine.py

```python
import json
from typing import Dict, Any, List, Literal
from langgraph.graph import StateGraph, END
from langgraph.checkpoint import MemorySaver
from core.agent_factory import agent_factory

class AgentState(Dict[str, Any]):
    """Agent状态类型"""
    pass

class WorkflowEngine:
    """工作流编排引擎"""
    
    def __init__(self):
        self._compiled_workflows = {}
    
    def get_workflow(self, workflow_config: Dict[str, Any]):
        """获取或构建工作流"""
        workflow_id = workflow_config.get("id")
        
        if workflow_id in self._compiled_workflows:
            return self._compiled_workflows[workflow_id]
        
        # 根据类型构建
        flow_type = workflow_config.get("flow_type", "supervisor")
        
        if flow_type == "supervisor":
            graph = self._build_supervisor_graph(workflow_config)
        elif flow_type == "sequential":
            graph = self._build_sequential_graph(workflow_config)
        elif flow_type == "graph":
            graph = self._build_custom_graph(workflow_config)
        else:
            raise ValueError(f"Unknown flow type: {flow_type}")
        
        compiled = graph.compile(checkpointer=MemorySaver())
        self._compiled_workflows[workflow_id] = compiled
        return compiled
    
    def _build_supervisor_graph(self, config: Dict[str, Any]):
        """构建监督者模式工作流"""
        graph = StateGraph(AgentState)
        
        # 获取主管Agent和工作Agent
        supervisor_config = config.get("supervisor_agent")
        worker_configs = config.get("worker_agents", [])
        
        if not supervisor_config:
            raise ValueError("Supervisor agent required for supervisor mode")
        
        # 添加主管节点
        supervisor_node = agent_factory.create_node(supervisor_config)
        graph.add_node("supervisor", supervisor_node)
        
        # 添加工作节点
        worker_nodes = {}
        for worker in worker_configs:
            worker_node = agent_factory.create_node(worker)
            node_name = f"worker_{worker['id']}"
            graph.add_node(node_name, worker_node)
            worker_nodes[worker['name']] = node_name
            # 工作节点执行完后返回主管
            graph.add_edge(node_name, "supervisor")
        
        # 条件路由
        def route_supervisor(state: AgentState) -> str:
            next_agent = state.get("next_agent", "end")
            if next_agent == "end" or next_agent not in worker_nodes:
                return "end"
            return worker_nodes[next_agent]
        
        graph.add_conditional_edges("supervisor", route_supervisor, {
            "end": END,
            **{node_name: node_name for node_name in worker_nodes.values()}
        })
        
        graph.set_entry_point("supervisor")
        return graph
    
    def _build_sequential_graph(self, config: Dict[str, Any]):
        """构建顺序执行工作流"""
        graph = StateGraph(AgentState)
        
        worker_configs = config.get("worker_agents", [])
        prev_node = None
        
        for i, worker in enumerate(worker_configs):
            node_name = f"step_{i}_{worker['name']}"
            node = agent_factory.create_node(worker)
            graph.add_node(node_name, node)
            
            if prev_node:
                graph.add_edge(prev_node, node_name)
            else:
                graph.set_entry_point(node_name)
            
            prev_node = node_name
        
        if prev_node:
            graph.add_edge(prev_node, END)
        
        return graph
    
    def _build_custom_graph(self, config: Dict[str, Any]):
        """构建自定义图工作流"""
        graph = StateGraph(AgentState)
        
        # 添加所有节点
        nodes = {}
        for agent in config.get("worker_agents", []):
            node_name = f"node_{agent['id']}"
            node = agent_factory.create_node(agent)
            graph.add_node(node_name, node)
            nodes[agent['id']] = node_name
        
        # 添加边
        edges = config.get("edges", [])
        for edge in edges:
            from_node = nodes.get(edge["from"])
            to_node = nodes.get(edge["to"])
            if from_node and to_node:
                graph.add_edge(from_node, to_node)
        
        # 设置入口点
        start_nodes = config.get("start_nodes", [])
        if start_nodes:
            graph.set_entry_point(nodes[start_nodes[0]])
        
        return graph

workflow_engine = WorkflowEngine()
```
